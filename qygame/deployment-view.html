@startuml
title 服务器物理视图 (非k8s部署)

    actor client
    actor 运维 as ops

    client --[hidden]-> ops
}

together{
    cloud gate_server{
        node gate
        node cadvisor_gate
    }
    cloud hot_server {
        node hot
        node cadvisor_hot
    }

    cloud monitor_server{
        node node_exporter
        node prometheus
        node grafana
    }

    'gate_server放在hot_server之上
    gate_server --[hidden]> hot_server
    hot_server --[hidden]> monitor_server
}

        node center

        node db

    }

    cadvisor_server -[hidden]u-> data
}
